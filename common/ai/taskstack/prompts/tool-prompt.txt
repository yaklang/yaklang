    "type": "object",
    "required": ["tool", "@action"],
    "additionalProperties": false,
    "properties": {
        "@action": {
            "const": "require-tool",
            "description": "标识当前操作的具体类型"
        },
        "tool": {
            "type": "string",
            "description": "您想要了解具体参数的工具名"
        }
    }
}

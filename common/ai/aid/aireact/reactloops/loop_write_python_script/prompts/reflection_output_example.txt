#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""HTTP GET Request Tool"""
import argparse
import urllib.request
import json

def make_request(url, timeout=30):
    req = urllib.request.Request(url)
    with urllib.request.urlopen(req, timeout=timeout) as response:
        return response.read().decode('utf-8')

def main():
    parser = argparse.ArgumentParser(description="Make HTTP GET requests")
    parser.add_argument("url", help="Target URL")
    parser.add_argument("--timeout", type=int, default=30, help="Request timeout")
    args = parser.parse_args()
    result = make_request(args.url, args.timeout)
    print(result)

if __name__ == "__main__":
    main()

## Common Errors to Avoid

Error 1: Calling write_script when code already exists
WRONG: {"@action": "write_script", ...}  // Code already exists!
RIGHT: {"@action": "modify_script", ...}  // Fix specific lines

Error 2: Replacing third-party imports to avoid dependencies
WRONG: Replace `import requests` with `import urllib.request` to avoid installing
RIGHT: Use bash to install: `pip3 install --user requests`

Error 3: Repeating the same action without new information
WRONG: Call check_and_install_dependencies 3 times without installing anything
RIGHT: Check once -> install via bash -> check again to verify
